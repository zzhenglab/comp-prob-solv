import numpy as np
from sklearn.linear_model import LinearRegression
X = np.array([[1.0],[2.0],[3.0],[4.0]])
y = np.array([2.1, 4.1, 6.2, 8.0])
model = LinearRegression().fit(X, y)
print(f'Slope = {model.coef_[0]:.2f}')
print(f'Intercept = {model.intercept_:.2f}')
